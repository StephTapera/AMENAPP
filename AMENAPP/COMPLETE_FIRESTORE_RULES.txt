rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Helper functions
    function isSignedIn() {
      return request.auth != null;
    }

    function isOwner(userId) {
      return isSignedIn() && request.auth.uid == userId;
    }

    // Email-verified writer: password accounts must have a verified address.
    // Social/OAuth accounts (Google, Apple, etc.) are implicitly trusted.
    function isVerifiedWriter() {
      return request.auth != null && (
        request.auth.token.email_verified == true ||
        request.auth.token.firebase.sign_in_provider != 'password'
      );
    }

    function isAuthoredByRequester(data) {
      return isSignedIn() && (
        ('authorId' in data && data.authorId == request.auth.uid) ||
        ('authorUid' in data && data.authorUid == request.auth.uid) ||
        ('userId' in data && data.userId == request.auth.uid)
      );
    }

    function isOwnedOrCreatedByRequester(data) {
      return isSignedIn() && (
        isAuthoredByRequester(data) ||
        ('ownerId' in data && data.ownerId == request.auth.uid) ||
        ('createdBy' in data && data.createdBy == request.auth.uid) ||
        ('adminIds' in data && request.auth.uid in data.adminIds)
      );
    }

    function hasOnlyChanged(allowedFields) {
      return request.resource.data.diff(resource.data).affectedKeys().hasOnly(allowedFields);
    }

    function hasNotChanged(protectedFields) {
      return !request.resource.data.diff(resource.data).affectedKeys().hasAny(protectedFields);
    }

    function isValidPostVisibility(data) {
      return !('visibility' in data) || data.visibility in [
        'everyone', 'followers', 'community', 'private', 'covenant',
        'Everyone', 'Followers', 'Community Only'
      ];
    }

    function isValidPostCategory(data) {
      return !('category' in data) || data.category in [
        'openTable', 'testimonies', 'prayer', 'tip', 'funFact', 'sync'
      ];
    }

    function hasNoClientModerationFields(data) {
      return !data.keys().hasAny([
        'moderationStatus',
        'publicationVisibility',
        'flaggedForReview',
        'removed',
        'trueSource',
        'isTestContent'
      ]);
    }

    function isSafePostCreate() {
      return isVerifiedWriter()
        && request.auth.uid == request.resource.data.authorId
        && isValidPostVisibility(request.resource.data)
        && isValidPostCategory(request.resource.data)
        && hasNoClientModerationFields(request.resource.data);
    }

    function isSafePostAuthorUpdate() {
      return isVerifiedWriter()
        && request.auth.uid == resource.data.authorId
        && request.resource.data.authorId == resource.data.authorId
        && (!('createdAt' in resource.data) || request.resource.data.createdAt == resource.data.createdAt)
        && hasNotChanged([
          'moderationStatus',
          'publicationVisibility',
          'flaggedForReview',
          'removed',
          'trueSource',
          'isTestContent'
        ]);
    }

    // ===== USERS COLLECTION =====
    match /users/{userId} {
      // Allow read for username lookup (needed for login)
      allow read: if true;

      // Unverified allowed: profile creation happens during onboarding, before verification
      allow create: if isSignedIn() && request.auth.uid == userId;

      // Allow users to update their own profile
      // ✅ IMPORTANT: Allow updating counts (needed for follow/unfollow)
      allow update: if isSignedIn() && (
        // User updating their own profile
        isOwner(userId) ||
        // Allow updating counts for follow/unfollow operations
        (request.resource.data.diff(resource.data).affectedKeys()
          .hasOnly(['followersCount', 'followingCount', 'updatedAt']))
      );

      // Only allow users to delete their own profile
      allow delete: if isSignedIn() && isOwner(userId);
    }

    // ===== POSTS COLLECTION =====
    match /posts/{postId} {
      // Keep signed-in reads for compatibility with existing collection queries.
      // Visibility filtering must still be enforced by feed queries and route guards.
      allow read: if isSignedIn();
      allow create: if isSafePostCreate();
      allow update: if isSafePostAuthorUpdate()
                    || (isSignedIn()
                        && hasOnlyChanged([
                          'amenCount',
                          'lightbulbCount',
                          'commentCount',
                          'repostCount',
                          'savesCount',
                          'sharesCount',
                          'prayTapsCount',
                          'encouragedCount'
                        ]));
      allow delete: if isVerifiedWriter()
                   && request.auth.uid == resource.data.authorId;

      match /amens/{userId} {
        allow read: if isSignedIn();
        // Reaction writes go through togglePostReaction Cloud Function.
        allow write: if false;
      }

      match /lightbulbs/{userId} {
        allow read: if isSignedIn();
        // Reaction writes go through togglePostReaction Cloud Function.
        allow write: if false;
      }

      match /reposts/{userId} {
        allow read: if isSignedIn();
        allow create, update: if isVerifiedWriter()
                             && request.auth.uid == userId
                             && request.resource.data.userId == request.auth.uid;
        allow delete: if isSignedIn() && request.auth.uid == userId;
      }

      match /comments/{commentId} {
        allow read: if isSignedIn();
        allow create: if isVerifiedWriter()
                     && isAuthoredByRequester(request.resource.data);
        allow update: if (isAuthoredByRequester(resource.data)
                          && hasNotChanged(['authorId', 'authorUid', 'userId', 'postId', 'createdAt']))
                      || (isSignedIn() && hasOnlyChanged(['replyCount', 'likeCount']));
        allow delete: if isAuthoredByRequester(resource.data);

        match /likes/{userId} {
          allow read: if isSignedIn();
          allow create, delete: if isSignedIn() && request.auth.uid == userId;
          allow update: if false;
        }
      }
    }

    // ===== TESTIMONIES COLLECTION =====
    match /testimonies/{testimonyId} {
      allow read: if isSignedIn();
      allow create: if isVerifiedWriter() && isAuthoredByRequester(request.resource.data);
      allow update: if isAuthoredByRequester(resource.data)
                   && hasNotChanged(['authorId', 'authorUid', 'userId', 'createdAt']);
      allow delete: if isAuthoredByRequester(resource.data);
    }

    // ===== PRAYERS COLLECTION =====
    match /prayers/{prayerId} {
      allow read: if isSignedIn();
      allow create: if isVerifiedWriter() && isAuthoredByRequester(request.resource.data);
      allow update: if isAuthoredByRequester(resource.data)
                   && hasNotChanged(['authorId', 'authorUid', 'userId', 'createdAt']);
      allow delete: if isAuthoredByRequester(resource.data);
    }

    // ===== COMMENTS COLLECTION =====
    match /comments/{commentId} {
      allow read: if isSignedIn();
      allow create: if isVerifiedWriter()
                   && isAuthoredByRequester(request.resource.data);
      allow update: if (isAuthoredByRequester(resource.data)
                        && hasNotChanged(['authorId', 'authorUid', 'userId', 'postId', 'createdAt']))
                    || (isSignedIn() && hasOnlyChanged(['replyCount', 'likeCount']));
      allow delete: if isAuthoredByRequester(resource.data);

      match /likes/{userId} {
        allow read: if isSignedIn();
        allow create, delete: if isSignedIn() && request.auth.uid == userId;
        allow update: if false;
      }
    }

    // ===== FOLLOWS COLLECTION =====
    match /follows/{followId} {
      // Authenticated reads support followers/following/mutuals queries.
      allow read: if isSignedIn();

      // The follower can create/delete their own edge only. No client updates.
      allow create: if isVerifiedWriter()
                   && request.auth.uid == request.resource.data.followerId
                   && request.resource.data.followingId is string
                   && request.resource.data.followingId != request.auth.uid;
      allow update: if false;
      allow delete: if isSignedIn()
                   && request.auth.uid == resource.data.followerId;
    }

    // ===== FOLLOWING SUBCOLLECTION =====
    match /following/{userId}/user_following/{followingId} {
      allow read: if isSignedIn();
      allow create, update, delete: if isVerifiedWriter() && isOwner(userId);
    }

    match /following/{userId}/user_followers/{followerId} {
      allow read: if isSignedIn();
      // Only the verified follower may write their own mirror doc.
      // Removing the profile-owner path prevents forging follower records.
      allow create, update, delete: if isVerifiedWriter() && isOwner(followerId);
    }

    // ===== CONVERSATIONS COLLECTION =====
    match /conversations/{conversationId} {
      function isParticipant() {
        return request.auth.uid in resource.data.participantIds;
      }

      function willBeParticipant() {
        return request.auth.uid in request.resource.data.participantIds;
      }

      function isAcceptedConversation() {
        return resource.data.conversationStatus == 'accepted';
      }

      function isPendingForUser() {
        return resource.data.conversationStatus == 'pending' &&
               request.auth.uid in resource.data.participantIds;
      }

      // Allow read if user is a participant (includes pending requests)
      allow read: if isSignedIn() && isParticipant();

      // Require verified email to start a new conversation
      allow create: if isVerifiedWriter() && willBeParticipant();

      // Allow update if user is a participant
      // This includes accepting requests, marking as read, updating unread counts
      allow update: if isSignedIn() && isParticipant();

      // Allow delete if user is a participant
      allow delete: if isSignedIn() && isParticipant();

      // Messages subcollection
      match /messages/{messageId} {
        // Allow read only if user is a participant in the parent conversation
        allow read: if isSignedIn() &&
                       request.auth.uid in get(/databases/$(database)/documents/conversations/$(conversationId)).data.participantIds;

        // Require verified email to send messages
        allow create: if isVerifiedWriter() &&
                         request.auth.uid in get(/databases/$(database)/documents/conversations/$(conversationId)).data.participantIds;

        // Allow update/delete only if user is the sender
        allow update, delete: if isSignedIn() &&
                                 request.auth.uid == resource.data.senderId;
      }

      // Typing indicators
      match /typing/{userId} {
        allow read: if isSignedIn();
        allow write: if isSignedIn() && isOwner(userId);
      }
    }

    // Note: Message requests are now handled within conversations collection
    // with conversationStatus field ('pending', 'accepted', 'blocked')
    // No separate messageRequests collection needed

    // ===== BLOCKED USERS =====
    match /blockedUsers/{blockId} {
      // Canonical top-level block docs are written by Cloud Functions.
      // Clients may read only relationships involving themselves.
      allow read: if isSignedIn()
                  && (blockId.matches(request.auth.uid + '_.*')
                      || blockId.matches('.*_' + request.auth.uid)
                      || resource.data.blockerId == request.auth.uid
                      || resource.data.blockedId == request.auth.uid);
      allow write: if false;
    }

    match /blockedUsers/{userId}/blocked/{blockedUserId} {
      // Legacy nested block list retained for compatibility; owner-only.
      allow read, write: if isSignedIn() && isOwner(userId);
    }

    match /users/{userId}/blockedUsers/{blockedUserId} {
      // Canonical per-user block mirror written by Cloud Functions. Owner read only.
      allow read: if isSignedIn() && isOwner(userId);
      allow write: if false;
    }

    // ===== SAVED POSTS =====
    match /savedPosts/{savedPostId} {
      // Supports legacy doc IDs shaped as `{uid}_{postId}` and owner field checks.
      allow read: if isSignedIn()
                  && (savedPostId.matches(request.auth.uid + '_.*')
                      || resource.data.userId == request.auth.uid);
      allow create, update: if isVerifiedWriter()
                            && request.resource.data.userId == request.auth.uid
                            && savedPostId.matches(request.auth.uid + '_.*');
      allow delete: if isSignedIn()
                    && (savedPostId.matches(request.auth.uid + '_.*')
                        || resource.data.userId == request.auth.uid);
    }

    match /savedPosts/{userId}/{document=**} {
      allow read: if isSignedIn() && isOwner(userId);
      allow write: if isVerifiedWriter() && isOwner(userId);
    }

    // ===== NOTIFICATIONS =====
    match /notifications/{notificationId} {
      // Users can read their own notifications
      allow read: if isSignedIn()
                 && request.auth.uid == resource.data.userId;

      // SECURITY: No client-side creates. All notification records are written
      // by Cloud Functions (Admin SDK) which bypass these rules. Allowing
      // client creates would let any user forge notifications for other users.
      allow create: if false;

      // Users can mark their own notifications as read / archive them.
      // They cannot modify actorId, type, or other server-set fields.
      allow update: if isSignedIn()
                   && request.auth.uid == resource.data.userId
                   && request.resource.data.diff(resource.data).affectedKeys()
                        .hasOnly(['read', 'seenAt', 'openedAt', 'dismissedAt', 'updatedAt']);

      // Users can delete their own notifications
      allow delete: if isSignedIn()
                   && request.auth.uid == resource.data.userId;
    }

    // ===== USER SUBCOLLECTIONS =====

    // Multi-device FCM tokens — owner read/write only.
    // Server can write via Admin SDK (bypasses rules).
    match /users/{userId}/devices/{deviceId} {
      allow read, write: if isSignedIn() && isOwner(userId);
    }

    // Notification preferences (canonical per-feature settings).
    // Owner read/write; server can override via Admin SDK.
    match /users/{userId}/settings/{document=**} {
      allow read, write: if isSignedIn() && isOwner(userId);
    }

    // In-app notification inbox records written by Cloud Functions.
    // Owner read only; only owner may mark read/dismissed.
    match /users/{userId}/notifications/{notificationId} {
      allow read: if isSignedIn() && isOwner(userId);
      // No client creates — Cloud Functions (Admin SDK) write these.
      allow create: if false;
      allow update: if isSignedIn()
                   && isOwner(userId)
                   && request.resource.data.diff(resource.data).affectedKeys()
                        .hasOnly(['read', 'seenAt', 'openedAt', 'dismissedAt', 'updatedAt']);
      allow delete: if isSignedIn() && isOwner(userId);
    }

    // ===== CANONICAL NOTIFICATION PIPELINE COLLECTIONS =====
    // These collections are managed exclusively by Cloud Functions (Admin SDK).
    // Clients may read their own records; no client writes allowed.

    match /notification_groups/{groupId} {
      // Recipient can read their own notification groups
      allow read: if isSignedIn()
                 && request.auth.uid == resource.data.recipientId;
      // Only clients may mark read/seen/dismissed; server writes everything else
      allow update: if isSignedIn()
                   && request.auth.uid == resource.data.recipientId
                   && request.resource.data.diff(resource.data).affectedKeys()
                        .hasOnly(['readAt', 'seenAt', 'openedAt', 'dismissedAt', 'updatedAt']);
      // No client creates or deletes — Admin SDK only
      allow create, delete: if false;
    }

    match /activity_events/{eventId} {
      // Recipient can read their own activity events
      allow read: if isSignedIn()
                 && request.auth.uid == resource.data.recipientId;
      // No client writes — Cloud Functions (Admin SDK) only
      allow write: if false;
    }

    match /pending_notifications/{recordId} {
      // No client access — internal Cloud Functions pipeline only
      allow read, write: if false;
    }

    match /notification_delivery_logs/{logId} {
      // No client access — internal audit log only
      allow read, write: if false;
    }

    match /notification_dead_letters/{recordId} {
      // No client access
      allow read, write: if false;
    }

    match /notification_metrics/{docId} {
      // No client access — server counters only
      allow read, write: if false;
    }

    // ===== COMMUNITIES =====
    match /communities/{communityId} {
      allow read: if isSignedIn();
      allow create: if isVerifiedWriter()
                   && isOwnedOrCreatedByRequester(request.resource.data);
      allow update, delete: if isVerifiedWriter()
                            && isOwnedOrCreatedByRequester(resource.data)
                            && hasNotChanged(['ownerId', 'createdBy', 'createdAt']);
    }

    // ===== REPOSTS =====
    match /reposts/{repostId} {
      allow read: if isSignedIn();
      allow create: if isVerifiedWriter()
                   && isAuthoredByRequester(request.resource.data);
      allow update: if false;
      allow delete: if isAuthoredByRequester(resource.data);
    }

    // ===== USER REPOSTS (Subcollection) =====
    match /user-reposts/{userId}/{document=**} {
      // Allow anyone authenticated to read user's reposts for profile view.
      allow read: if isSignedIn();
      // Only the verified owner can write their repost index.
      allow write: if isVerifiedWriter() && isOwner(userId);
    }

    // ===== POST INTERACTIONS =====
    match /postInteractions/{interactionId} {
      allow read: if isSignedIn();
      allow create: if isVerifiedWriter()
                   && isAuthoredByRequester(request.resource.data);
      allow update: if isAuthoredByRequester(resource.data);
      allow delete: if isAuthoredByRequester(resource.data);
    }
  }
}
